    lower_classic, upper_classic = mean - 1.96 * se, mean + 1.96 * se

    bootstrap_means = np.mean(bootstrap_samples, axis=1)
    lower_bootstrap, upper_bootstrap = np.percentile(bootstrap_means, [2.5, 97.5])
